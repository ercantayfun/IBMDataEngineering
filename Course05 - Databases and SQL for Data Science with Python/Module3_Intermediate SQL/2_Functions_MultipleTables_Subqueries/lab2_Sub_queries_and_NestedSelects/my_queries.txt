SELECT AVG(SALARY) 
FROM (SELECT SALARY 
      FROM EMPLOYEES 
      ORDER BY SALARY 
      LIMIT 5) AS SALARY_TABLE;

SELECT * 
FROM EMPLOYEES 
WHERE FROM_DAYS(DATEDIFF(CURRENT_DATE,B_DATE)) > 
    (SELECT AVG(FROM_DAYS(DATEDIFF(CURRENT_DATE,B_DATE))) 
    FROM EMPLOYEES);


SELECT EMPL_ID, YEAR(FROM_DAYS(DATEDIFF(CURRENT_DATE, START_DATE))), 
    (SELECT AVG(YEAR(FROM_DAYS(DATEDIFF(CURRENT_DATE, START_DATE)))) 
    FROM JOB_HISTORY)
FROM JOB_HISTORY;
