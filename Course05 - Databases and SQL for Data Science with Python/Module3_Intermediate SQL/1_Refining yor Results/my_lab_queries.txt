SELECT F_NAME, L_NAME
FROM EMPLOYEES
WHERE F_NAME LIKE 'S%';

SELECT * FROM EMPLOYEES ORDER BY B_DATE DESC;

SELECT DEP_ID, AVG (SALARY) AS AVG_SAL 
FROM EMPLOYEES 
GROUP BY DEP_ID 
HAVING AVG_SAL >= 60000;

SELECT DEP_ID, AVG (SALARY) AS AVG_SAL 
FROM EMPLOYEES 
GROUP BY DEP_ID 
HAVING AVG_SAL >= 60000
ORDER BY AVG_SAL DESC;
